begin

    {\n
        Store (One, \\_SB.PCI0.LPCB.AG3E)\n
        Store (Zero, SLPE)\n
        Sleep (0x10)\n
    }\n
    Else\n
    {\n
        Store (Zero, \\_SB.PCI0.LPCB.AG3E)\n
%1
    }
end;
